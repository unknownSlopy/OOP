#include <iostream>
#include <vector>
#include <string>
#include <functional>

using namespace std;

class Proizvod {
private:
	string naziv;
	int kolicina;
	double cijena;
public:
	Proizvod(string _naziv, int _kolicina, double _cijena) : naziv(_naziv), kolicina(_kolicina), cijena(_cijena) {};

	string getNaziv() const {
		return naziv;
	}

	int getKolicina() const {
		return kolicina;
	}

	double getCijena() const {
		return cijena;
	}


};


vector<Proizvod*> nadji(vector<Proizvod>& proizvodi, function<bool(Proizvod*)> kriterij) {
	vector<Proizvod*> rezultat;
	for (int i = 0; i < proizvodi.size(); i++) {
		if (kriterij(&proizvodi[i])) {
			rezultat.push_back(&proizvodi[i]);

		}
	}

	return rezultat;

}


int main() {
	vector<Proizvod> v = { Proizvod("Mobitel",221,760.5),
						  Proizvod("TV",24,2017.0),
						  Proizvod("Tablet",85,3155.65),
						  Proizvod("Laptop",7,9200) };

	// nadi sve proizvode s cijenom vecom od 1000


	/* ispisuje

	TV 24 2017
	Tablet 85 3155.65
	Laptop 7 9200


	*/
	vector<Proizvod*> r = nadji(v, [](Proizvod* s) {return s->getCijena() > 1000; });

	for (Proizvod* p : r) {
		cout << p->getNaziv() << " " << p->getKolicina() << " " << p->getCijena() << endl;
	}

	return 0;
}
