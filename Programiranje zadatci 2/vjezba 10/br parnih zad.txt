#include <iostream>
#include <vector>

using namespace std;

int main() {

	vector<int> brojevi = { 1, 4, 5, 6 ,7 ,3, 5, 12, 87, 55, 22 };

	int brParnih = 0;

	auto parni = [&brParnih](vector<int>& v) {
		for (int i = 0; i < v.size(); i++)
			if (v[i] % 2 == 0)
				brParnih++;
	};

	parni(brojevi);

	cout << brParnih << endl;

	return 0;
}
