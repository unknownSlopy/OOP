#include <iostream>
#include <vector>

using namespace std;

int main() {

    vector<int> brojevi = { 1, 4, 5, 7, 3, 6, 12, 65, 32, 8, 87, 55, 23, 22, 1, 1, 433, 66, 7, 433, 3, 32, 76, 8, 72, 256, 42 };
    int increment = 3;

    [](vector<int>& v, int ink) {
        for (int i = 0; i < v.size(); i++) {
            v[i] += ink;
        }
    }(brojevi, increment);

    for (const int& broj : brojevi) {
        cout << broj << ",";
    }

    return 0;
}
