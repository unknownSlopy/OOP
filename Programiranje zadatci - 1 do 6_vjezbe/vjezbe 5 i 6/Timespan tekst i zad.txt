#include<iostream>

using namespace std;

class TimeSpan {
private:
	int sati, minute, sekunde;

public:
	friend ostream& operator << (ostream& izlaz, TimeSpan t);
	
	TimeSpan(int _sati, int _minute, int _sekunde) : sati(_sati), minute(_minute), sekunde(_sekunde) {}

	/*TimeSpan operator + (const TimeSpan& t) {

		sati = sve_sek / 3600;
		minute = (sve_sek % 3600) / 60;
		sekunde = sve_sek % 60;
		return TimeSpan(sati, minute, sekunde);
	}*/

	TimeSpan operator + (const TimeSpan& t) {
		int sve_sek = sekunde + t.sekunde + 60 * (minute + t.minute) + 3600 * (sati + t.sati);
		minute = sve_sek / 60;
		sekunde = sve_sek % 60;
		sati = minute / 60;
		minute = minute % 60;
		return TimeSpan(sati, minute, sekunde);
	}

	bool operator < (const TimeSpan& t) {
		return bool(sati < t.sati || minute < t.minute || sekunde < t.sekunde);
	}
	

	bool operator == (const TimeSpan& t) {
		return bool(sati == t.sati && minute == t.minute && sekunde == t.sekunde);
	}

	operator int() {
		return sekunde + 60 * minute + 3600 * sati;
	}
};

ostream& operator << (ostream& izlaz, TimeSpan t) {
	izlaz << t.sati << ":" << t.minute << ":" << t.sekunde << endl;
	return izlaz;
}
