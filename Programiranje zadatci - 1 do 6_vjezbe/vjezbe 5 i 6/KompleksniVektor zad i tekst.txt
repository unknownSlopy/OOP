#include <iostream>
#include <vector>

using namespace std;

class Kompleksni {
private:
	double re, im;

public:
	static int br;
	Kompleksni(double _re, double _im) : re(_re), im(_im) {
		br++;
	}
	
	Kompleksni() {
		br++;
	}

	Kompleksni(const Kompleksni& other) : re(other.re), im(other.im) {
		br++;
	}
	~Kompleksni() {
		br--;
	}
	static int brojac() {
		return br;
	}

	friend class KompleksniVektor;
	friend ostream& operator << (ostream& izlaz, Kompleksni z);
	friend istream& operator >>(istream& ulaz, Kompleksni z);
	
	Kompleksni& operator ++() {
		++re;
		++im;
		return *this;
	}

	Kompleksni operator ++(int) {
		Kompleksni pom = *this;
		re++;
		im++;
		return pom;
	}

};
int Kompleksni::br = 0;


class KompleksniVektorException : public exception{
public:
	const char* ispis() const noexcept {
		return "Indeks je negativan, nemere!";
	}
};

ostream& operator << (ostream& izlaz, Kompleksni z) {
	izlaz << z.re << " + " << z.im << "i";
	return izlaz;
}

istream& operator >>(istream& ulaz, Kompleksni z) {
	ulaz >> z.re >> z.im;
	return ulaz;
}

class KompleksniVektor {
private:
	vector<Kompleksni> kompleksni_vektor;

public:
	friend ostream& operator <<(ostream& izlaz, KompleksniVektor vektor);
	friend Kompleksni GetElement(const KompleksniVektor& kompleksni, int indeks);
	// friend KompleksniVektor operator ==(KompleksniVektor vektor1, KompleksniVektor vektor2);
	
	void dodaj(const Kompleksni& kompleksni) {
		 kompleksni_vektor.push_back(kompleksni);
	 }

	 KompleksniVektor operator ++() {
		 for (auto& kompleksni : kompleksni_vektor) {
			 ++kompleksni;
		 }
		 return *this;
	 }

	 KompleksniVektor operator ++(int) {
		 KompleksniVektor pom = *this;
		 ++(*this);
		 return pom;
	 }
};

Kompleksni GetElement(const KompleksniVektor& kompleksni, int indeks) {
	if (indeks < 0) {
		throw KompleksniVektorException();
	}
	kompleksni.kompleksni_vektor.at(indeks);
}
