#include<iostream>
#include<vector>
#include<string>

using namespace std;

class Zaposlenik {
private:
	string ime;
	string prezime;

public:

	void setIme(string _ime) {
		ime = _ime;
	}

	void setPrezime(string _prezime) {
		prezime = _prezime;
	}

	string getIme() const {
		return ime;
	}

	string getPrezime() const {
		return prezime;
	}

	Zaposlenik(string _ime, string _prezime) : ime(_ime), prezime(_prezime) {
		cout << "Zaposlenik constructor called" << endl;
	}

	~Zaposlenik() {
		std::cout << "Zaposlenik destructor called" << endl;
	}
};

class Firma {
private:
	string naziv;
	string OIB;
	string adresaSjedista;

public:

	vector<Zaposlenik> vZaposlenici;
 
	void setNAziv(string _naziv) {
		if (_naziv.size() >= 5) {
			naziv = _naziv;
		}
		else {
			cout << "Unijeli ste naziv manji od 5 znakova." << endl;
		}
	}

	void setOIB(string _OIB) {
		if (_OIB.size() == 11) {
			OIB = _OIB;
		}
		else {
			cout << "Unijeli ste OIB koji nemaa tocno 11 znakova." << endl;
		}
	}

	void setAdresaSjedista(string _adresaSjedista) {
		adresaSjedista = _adresaSjedista;
	}

	string getNAziv() const{
		return naziv;
	}

	string getOIB() const {
		return OIB;
	}

	string getAdresaSjedista() const {
		return adresaSjedista;
	}

	void dodajZaposlenika1(Zaposlenik z) {
		vZaposlenici.push_back(z);
	}

	void dodajZaposlenika2(Zaposlenik& z) {
		vZaposlenici.push_back(z);
	}

	void dodajZaposlenika3(Zaposlenik *z) {
		vZaposlenici.push_back(*z);
	}

	void IspisSvihZaposlenika() {
		for (int i = 0; i < vZaposlenici.size(); i++) {
			cout << "Zaposlenik: " << vZaposlenici[i].getIme() << " " <<
				vZaposlenici[i].getPrezime() << endl;
		}
	}

	Zaposlenik nadjiZaposlenika1(string _ime, string _prezime) {
		for (int i = 0; i < vZaposlenici.size(); i++) {
			if (vZaposlenici[i].getIme() == _ime && vZaposlenici[i].getPrezime() == _prezime)
				return vZaposlenici[i];
		}
	}

	Zaposlenik* nadjiZaposlenika2(string _ime, string _prezime) {
		for (int i = 0; i < vZaposlenici.size(); i++) {
			if (vZaposlenici[i].getIme() == _ime && vZaposlenici[i].getPrezime() == _prezime)
				return &vZaposlenici[i];
		}
	}

	Zaposlenik& nadjiZaposlenika3(string _ime, string _prezime) {
		for (int i = 0; i < vZaposlenici.size(); i++) {
			if (vZaposlenici[i].getIme() == _ime && vZaposlenici[i].getPrezime() == _prezime)
				return vZaposlenici[i];
		}
	}
};

int main() {

	Zaposlenik z1("John", "Doe");
	Zaposlenik z2("Pero", "Peric");

	Firma firma;

	firma.dodajZaposlenika1(z1);
	firma.dodajZaposlenika1(z2);

	Zaposlenik* z3 = firma.nadjiZaposlenika2("John", "Doe");
	if (z3) {
		z3->setIme("Matija");
		z3->setPrezime("Hrvatko");
	}
	else cout << "Nismo pronasli zaposlenika." << endl;

	Zaposlenik& z4 = firma.nadjiZaposlenika3("Pero", "Peric");
	if (&z4) {
		z4.setIme("Ivana");
		z4.setPrezime("Horvat");
	}
	else cout << "Nismo pronasli zaposlenika." << endl;

	firma.IspisSvihZaposlenika();
	
	return 0;
}
